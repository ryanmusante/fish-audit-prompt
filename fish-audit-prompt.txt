STEP 1: DATA COLLECTION

  ```bash
  F=/home/claude/audit-src/script.fish
  D=/home/claude/audit-data
  O=/home/claude/audit-findings.txt
  mkdir -p "$D"
  # ── Resume guard: preserve partial findings from interrupted runs ──
  if [ -s "$O" ]; then
    mv "$O" "${O%.txt}-$(date +%s).bak"
    echo "[INFO] Previous findings backed up to ${O%.txt}-*.bak"
  fi
  : > "$O"

  # ── Python3 helper for function body extraction (primary; fish -c fails on some versions) ──
  _py_extract() {
    python3 -c "
import re, sys
target = sys.argv[2]; F = sys.argv[1]
lines = open(F).readlines()
cap = False; depth = 0
block_re = re.compile(r'^\s*(function |if |for |while |switch |begin\b)')
for line in lines:
    s = line.strip()
    if not cap:
        if re.match(r'^function ' + re.escape(target) + r'([ \-]|$)', line):
            cap = True; depth = 1
        continue
    if not s.startswith('else') and block_re.match(line):
        depth += 1
    if s == 'end':
        depth -= 1
        if depth == 0: break
        print(line.rstrip())
    else:
        print(line.rstrip())
" "$1" "$2"
  }
  export -f _py_extract

  # ── Centralized skip list for message/output functions (used by DRY GATE + BARE COREUTILS) ──
  MSG_SKIP='_run|_log|_dry|_fail|_err|_warn|_ok|_info|_echo|_msg|_ask'
  export MSG_SKIP

  # ── GROUP A1: P1-3 syntax, routing, errors ──
  (
    echo "=== BASHISMS ===" && rg -n '\$\(|export |&&|\|\||set -e |functions -e|unset |local ' "$F" | head -30

    echo "=== STRING OPS ===" && rg -n 'string (match|replace|split|join|trim)' "$F" | rg -v -- '--' | head -30
    echo "=== ARGPARSE ===" && rg -n 'argparse|argv' "$F" | head -30
    echo "=== ROUTING ===" && rg -n '>&2|>/dev/null|stderr|stdout' "$F" | head -50
    echo "=== NO_COLOR ===" && rg -n 'NO_COLOR|isatty|tput|test -t' "$F" | head -20
    echo "=== EXIT/RETURN ===" && rg -n 'exit [0-9]|return [0-9]|exit$' "$F" | head -30
    echo "=== STATUS ===" && rg -n '\$status|\$pipestatus' "$F" | head -30
    echo "=== SIGNALS ===" && rg -n 'trap|--on-signal|--on-event|fish_exit' "$F" | head -20
    echo "=== CLEANUP ===" && rg -n 'mktemp|rm -f|rm -rf|cleanup|_do_cleanup' "$F" | head -30
  ) > "$D/p1-3.txt" 2>&1; echo "GROUP_A1_EXIT=$?" >> "$D/p1-3.txt" &

  # ── GROUP A2: P4-5 vars, security ──

    echo "=== SCOPE ===" && rg -n 'set -[glx]|set --erase|set --global|set --local' "$F" | head -50
    echo "=== FOR LOOPS ===" && rg -n 'for .* in ' "$F" | head -30
    echo "=== INJECTION ===" && rg -n 'eval|source ' "$F" | head -20
    echo "=== CHK_GREP IMPL ===" && _py_extract "$F" "_chk_grep" | rg 'grep' | head -10
    echo "=== CREDS ===" && rg -ni 'password|passphrase|credential|secret|WIFI_PASS' "$F" | head -20
    echo "=== PERMS ===" && rg -n 'chmod|chown|0600|0644|0700|0755' "$F" | head -30
    echo "=== SUDO ===" && rg -n 'sudo |_IS_ROOT' "$F" | head -30
    echo "=== EXT CMDS ===" && rg -n 'command -q|command -v' "$F" | head -30
    echo "=== NUMERIC VAR INIT ===" && python3 -c "
import subprocess, sys; F = sys.argv[1]
# Excludes Fish builtins (\$status) and script globals (MAX_LOGS, _DIAG_ISSUES) — always initialized by runtime/top-level set -g
for v in 'errors total_err line_count ret idx_a idx_b perm_checked dir_checked verify_status failed code steps_count reload_wait wifi_retry next_i next_i2 _bl_loaded _log_count bad_perms dir_bad fixed_count i install_status mask_count missing mode_count pac_managed_bad pad perm_bad read_ok _boot_rc _flock_rc case_count checked record_idx'.split():

" "$F" 2>&1
  ) > "$D/p4-5.txt" 2>&1; echo "GROUP_A2_EXIT=$?" >> "$D/p4-5.txt" &

    echo "=== VERSION ===" && rg -n 'VERSION|SYSTEM_DEST|USER_DEST|SERVICE_DEST|get_file_content|MASK' "$F" | head -30
    echo "=== DISPATCH ===" && rg -n 'do_\w+ \$|do_\w+ "\$' "$F" | rg -v 'function |#|_log' | head -20
    echo "=== SYSCTL ===" && rg -n 'sysctl|BBR|tcp_fastopen|default_qdisc|inotify' "$F" | head -30

  # ── P13 analysis reads p14.txt for 11 shared patterns)       ──

  fi
  if [ ! -s "$D/p13.txt" ]; then
    echo "[INFO] p13.txt empty or missing — P13 gap analysis checks will be incomplete" >> "$O"
  fi

  # ── SIZE GATE: truncate any file over 500 lines ──
  for f in "$D"/*.txt; do
    lines=$(wc -l < "$f")

        p8-9)   phases="P8-P9 (architecture,patterns)" ;;
        p10-11) phases="P10-P11 (docs,testing)" ;;
        p13)    phases="P13 (gap analysis)" ;;
        p14)    phases="P14-P15 (v3.1.0-v3.7.1 specifics, deep checks)" ;;
        *)      phases="unknown ($fname)" ;;
      esac
      echo "WARNING: $f has $lines lines, truncating to 500" >&2
      echo "[INFO] SIZE GATE: $f ($phases) truncated from $lines to 500 lines — affected phase checks may have incomplete data" >> "$O"
      # ── Recovery manifest: list lost sections for targeted re-collection ──
      grep '^=== .* ===$' "$f" > "$D/${fname}-full-sections.txt"
      head -500 "$f" > "$f.tmp" && mv "$f.tmp" "$f"
      grep '^=== .* ===$' "$f" > "$D/${fname}-kept-sections.txt"
      comm -23 <(sort "$D/${fname}-full-sections.txt") <(sort "$D/${fname}-kept-sections.txt") > "$D/${fname}-lost.txt"

      fi
    fi
  done

  # ── EMPTY SECTION GUARD: detect sections with 0 data lines ──
  for f in "$D"/*.txt; do
    while IFS= read -r header; do
      section=$(echo "$header" | sed 's/^=== //;s/ ===$//')
      # Get line count between this header and the next (or EOF)
      linenum=$(grep -nF "$header" "$f" | head -n 1 | cut -d: -f1)
      nextnum=$(awk -v start="$linenum" 'NR>start && /^=== .* ===$/{print NR; exit}' "$f")
      : "${nextnum:=$(wc -l < "$f")}"
      content_lines=$((nextnum - linenum - 1))
      if [ "$content_lines" -le 0 ]; then
        echo "[INFO] EMPTY SECTION: $section in $(basename "$f") — rg may have failed silently" >> "$O"
      fi
    done < <(grep '^=== .* ===$' "$f")
  done

  echo "=== DATA COLLECTION COMPLETE ==="
  wc -l "$D"/*.txt
  ```

  Partial: save findings, mark SKIPPED phases, add COVERAGE line,
  still produce spec, verdict "incomplete — N/15 phases completed".

STEP 3: SPEC FILE

  SEVERITY: CRITICAL (data loss/security) | HIGH (functional bug) |
  MED (correctness) | LOW (style) | INFO (observation)

  Per finding: F-N | Severity | Phase | Checks | Lines | Category | Detail | Fix

  ```
  FIXES SPEC: <script> <old-ver> → <new-ver>
  Date: YYYY-MM-DD | Findings: N (N CRIT, N HIGH, N MED, N LOW, N INFO)

  --- FIX INSTRUCTIONS ---
  F-1  [SEV]  <summary>
       Checks: N,N  Lines: N,N  Category: <checklist item>
       Before: <exact text>
       After:  <exact replacement>
       Verify: <command>

  --- INFO (no action) ---
  I-1  [INFO]  <desc>  Checks: N  Lines: N

  --- VERSION ---
  Bump: <old> → <new>
  Changelog: v<new> <summary>.

  --- VERIFICATION ---
  1. fish --no-execute script.fish
  2. rg '<old-version>' script.fish README.md  (expect 0)
  3. rg '<new-version>' script.fish README.md  (expect N)
  ```

  Before/After must be exact and unique within the file (required for
  str_replace). If not unique, include enough surrounding context to
  disambiguate. Spec self-contained. Version bump if any actionable
  fixes. INFO listed separately.

  UNIQUENESS VERIFICATION (run DURING and AFTER spec generation):
  Before adding each Before/After pair, verify uniqueness first:
  ```bash
  verify_unique() {
    local pat="$1"
    local c=$(rg -cF -- "$pat" /home/claude/audit-src/script.fish 2>/dev/null || echo 0)
    if [ "$c" != "1" ]; then
      echo "NON-UNIQUE ($c hits): $pat -- expand context"
      return 1
    fi
  }
  ```
  If non-unique (0 or 2+ hits), expand Before/After with surrounding
  lines until unique. Post-write full check:
  ```bash
  rg -o 'Before: .*' fixes-spec-*.txt | while IFS= read -r line; do
    pat="${line#*Before: }"
    c=$(rg -cF -- "$pat" /home/claude/audit-src/script.fish 2>/dev/null || echo 0)
    [ "$c" != "1" ] && echo "NON-UNIQUE ($c hits): $pat"
  done
  ```
